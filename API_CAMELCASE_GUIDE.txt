async def list_documents(self, skip: int = 0, limit: int = 50):
    """获取文档列表"""

            "progress": doc["progress"],
            "message": doc["message"],
            "createdAt": doc["created_at"],  # 改为createdAt
            "updatedAt": doc["updated_at"]   # 改为updatedAt
        })

    # 排序和分页

## 🔧 完整修改方案

### 方案1: 使用辅助函数统一转换

**创建转换函数：**

```python
def to_camel_case(snake_str: str) -> str:
    """将蛇形命名转为小驼峰"""
    components = snake_str.split('_')
    return components[0] + ''.join(x.title() for x in components[1:])

def convert_keys_to_camel(data: dict) -> dict:
    """递归转换字典的键为小驼峰"""
    if isinstance(data, dict):
        return {to_camel_case(k): convert_keys_to_camel(v) for k, v in data.items()}
    elif isinstance(data, list):
        return [convert_keys_to_camel(item) for item in data]
    else:
        return data
```

**在API返回前使用：**

```python
@router.get("/list")
async def list_documents(skip: int = 0, limit: int = 50):
    documents = await document_task_service.list_documents(skip=skip, limit=limit)
    total = await document_task_service.count_documents()

    # 转换为小驼峰
    return convert_keys_to_camel({
        "documents": documents,
        "total": total,
        "skip": skip,
        "limit": limit
    })
```
