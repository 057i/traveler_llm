### 问题1: MinerU完整输出包的保存

#### 当前实现分析
**MinerU API返回：**
```python
# mineru_client.py:216
if state == "done":
    markdown_url = result["data"]["markdown_url"]  # 只有MD文件的URL
    md_response = requests.get(markdown_url, timeout=60)
    markdown_content = md_response.text
    return markdown_content  # 只返回文本
```

**问题：**
- MinerU API只返回`markdown_url`（MD文件下载链接）
- 没有返回完整的输出包（图片、表格等）
- 当前只下载了Markdown文本

#### 需要的完整实现

**步骤1: 修改MinerU客户端，获取完整结果**

MinerU API响应可能包含：
```json
{
  "data": {
    "state": "done",
    "markdown_url": "https://...",
    "result_url": "https://...",  // 完整输出包的ZIP下载链接
    "images": [...],
    "tables": [...]
  }
}
```

**步骤2: 下载并解压完整输出包**

```python
# mineru_client.py
def _poll_result(self, task_id: str) -> Optional[Dict[str, Any]]:
    """返回完整结果，包含MD和文件包路径"""
    if state == "done":
        result_data = result["data"]

        # 1. 下载Markdown文本
        markdown_url = result_data.get("markdown_url")
        markdown_content = requests.get(markdown_url).text

        # 2. 检查是否有完整输出包
        result_url = result_data.get("result_url")
        output_dir = None

        if result_url:
            # 下载ZIP包
            import tempfile
            import zipfile

            # 创建临时目录
            output_dir = tempfile.mkdtemp(prefix=f"mineru_{task_id}_")
            zip_path = os.path.join(output_dir, "result.zip")

            # 下载ZIP
            logger.info(f"[MinerU] 下载完整输出包...")
            zip_response = requests.get(result_url, timeout=120)
            with open(zip_path, 'wb') as f:
                f.write(zip_response.content)

            # 解压
            logger.info(f"[MinerU] 解压输出包...")
            with zipfile.ZipFile(zip_path, 'r') as zip_ref:
                zip_ref.extractall(output_dir)

            os.remove(zip_path)  # 删除ZIP文件
            logger.success(f"[MinerU] 输出包已解压到: {output_dir}")

        # 返回完整结果
        return {
            "markdown_content": markdown_content,
            "output_directory": output_dir,
            "has_output_package": output_dir is not None
        }
```

**步骤4: 修改MinIO节点上传完整输出包**

```python
# node_minio.py
async def save_to_minio(state: DocumentProcessingState):
    # 1. 上传原始PDF
    origin_filename = f"{name_without_ext}_origin{file_ext}"
    origin_object_name = f"{name_without_ext}/{origin_filename}"
    minio.save_file(file_path, origin_object_name)

    # 2. 上传MinerU完整输出包
    mineru_output_dir = state.get('mineru_output_dir')
    if mineru_output_dir and os.path.exists(mineru_output_dir):
        logger.info(f"[MinIO] 上传MinerU输出包...")

        # 遍历输出目录，上传所有文件
        for root, dirs, files in os.walk(mineru_output_dir):
            for filename in files:
                local_file = os.path.join(root, filename)

                # 计算相对路径
                rel_path = os.path.relpath(local_file, mineru_output_dir)

                # MinIO对象名称
                object_name = f"{name_without_ext}/mineru/{rel_path}"

                # 上传文件
                minio.save_file(local_file, object_name)
                logger.info(f"[MinIO] 已上传: {object_name}")

        logger.success(f"[MinIO] MinerU输出包上传完成")

        # 清理临时目录
        import shutil
        shutil.rmtree(mineru_output_dir)
        logger.info(f"[MinIO] 已清理临时目录")

    # 3. 单独上传Markdown文本（方便访问）
    markdown_text = state.get('parsed_text')
    if markdown_text:
        import tempfile
        with tempfile.NamedTemporaryFile(mode='w', encoding='utf-8', suffix='.md', delete=False) as f:
            f.write(markdown_text)
            temp_md_path = f.name

        md_object_name = f"{name_without_ext}/{name_without_ext}.md"
        minio.save_file(temp_md_path, md_object_name)
        os.remove(temp_md_path)
        logger.success(f"[MinIO] Markdown已上传: {md_object_name}")
```

```python
# documents.py
@router.delete("/{doc_id}")
async def delete_document(doc_id: str):
    """
    删除文档及相关数据

    清理范围：
    1. MinIO中的完整文件夹
    2. 本地临时文件
    3. Redis任务记录
    4. Neo4j图节点（可选）
    5. Milvus向量（可选）
    """
    try:
        logger.info(f"[Delete] 开始删除文档: {doc_id}")

        # 1. 获取文档信息
        doc_info = await document_task_service.get_document(doc_id)
        if not doc_info:
            raise HTTPException(status_code=404, detail="文档不存在")

        filename = doc_info.get('filename', '')
        minio_directory = doc_info.get('minio_directory', '')

        # 2. 删除MinIO文件夹
        if minio_directory:
            try:
                from app.core.minio_client import get_minio_client
                minio = get_minio_client()

                # 列出文件夹中的所有文件
                files = minio.list_files(prefix=f"{minio_directory}/")

                logger.info(f"[Delete] 删除MinIO文件夹: {minio_directory}/ ({len(files)}个文件)")

                # 删除所有文件
                for file_name in files:
                    minio.delete_file(file_name)
                    logger.debug(f"[Delete] 已删除MinIO文件: {file_name}")

                logger.success(f"[Delete] MinIO文件夹已清理")

            except Exception as e:
                logger.warning(f"[Delete] MinIO清理失败: {e}")

        # 3. 删除本地文件
        file_path = doc_info.get('file_path', '')
        if file_path and os.path.exists(file_path):
            os.remove(file_path)
            logger.info(f"[Delete] 本地文件已删除: {file_path}")

        # 4. 删除Neo4j节点（可选）
        if filename:
            try:
                from app.core.neo4j_client import get_neo4j_client
                neo4j = get_neo4j_client()

                # 删除该文档的所有Destination节点
                query = """
                MATCH (n:Destination {filename: $filename})
                DETACH DELETE n
                RETURN count(n) as deleted_count
                """
                result = neo4j.query(query, {"filename": filename})
                deleted_count = result[0]['deleted_count'] if result else 0

                logger.info(f"[Delete] Neo4j节点已删除: {deleted_count}个")

            except Exception as e:
                logger.warning(f"[Delete] Neo4j清理失败: {e}")

        # 5. 删除Milvus向量（可选）
        if filename:
            try:
                from app.core.milvus_hybrid_client import get_milvus_hybrid_client
                milvus = get_milvus_hybrid_client()

                # 删除该文档的所有向量
                expr = f'filename == "{filename}"'
                milvus.collection.delete(expr)

                logger.info(f"[Delete] Milvus向量已删除")

            except Exception as e:
                logger.warning(f"[Delete] Milvus清理失败: {e}")

        # 6. 删除Redis记录
        success = await document_task_service.delete_document(doc_id)
        if success:
            logger.success(f"[Delete] 文档删除完成: {doc_id}")

        return {
            "message": "删除成功",
            "doc_id": doc_id,
            "filename": filename
        }

    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"[Delete] 删除失败: {e}")
        raise HTTPException(status_code=500, detail=f"删除失败: {str(e)}")
```

### 问题3: JSON解析错误 - 切分问题

#### 问题诊断

**当前流程：**
```
1. PDF解析 → parsed_text (Markdown文本)
2. 文本分块 → chunks (切分后的文本块)
3. 实体提取 → 对每个chunk调用Qwen提取实体
```

**查看当前切分逻辑：**
```python
# node_chunking.py:17
parsed_text = state.get('parsed_text', '')  # ✓ 正确：获取Markdown

# node_chunking.py:31
chunks = splitter.split_text(parsed_text)  # ✓ 正确：切分Markdown
```

**结论：** 切分逻辑是**正确的**，在切分Markdown文本，不是JSON。

#### 真正的问题：Qwen返回格式不规范

**错误分析：**
```
JSON parse failed for chunk 9: Extra data: line 134 column 3 (char 3142)
```

这表示Qwen返回了：
```json
{"destinations": [...]}
额外的文本内容
```

或者：
```json
{"destinations": [...]}
{"destinations": [...]}  // 第二个JSON对象
```

#### 解决方案：增强JSON解析

```python
# node_entity_extraction.py
import json
from json import JSONDecoder

def extract_json_objects(text: str) -> list:
    """从文本中提取所有JSON对象"""
    decoder = JSONDecoder()
    objects = []
    idx = 0

    while idx < len(text):
        text_part = text[idx:].lstrip()
        if not text_part:
            break
        try:
            obj, end_idx = decoder.raw_decode(text_part)
            objects.append(obj)
            idx += end_idx + len(text[idx:]) - len(text_part)
        except json.JSONDecodeError:
            break

    return objects

def clean_json_response(text: str) -> str:
    """清理JSON响应"""
    text = text.strip()

    # 移除markdown代码块
    if text.startswith("```json"):
        text = text[7:]
    if text.startswith("```"):
        text = text[3:]
    if text.endswith("```"):
        text = text[:-3]

    # 提取第一个{到最后一个}
    start = text.find('{')
    end = text.rfind('}')
    if start != -1 and end != -1 and end > start:
        text = text[start:end+1]

    return text.strip()

# 在实体提取时使用
try:
    # 尝试标准解析
    result = json.loads(response_text)
except json.JSONDecodeError as e:
    logger.warning(f"[Extract] Standard JSON parse failed: {e}")

    # 方案1: 清理文本后重试
    try:
        cleaned_text = clean_json_response(response_text)
        result = json.loads(cleaned_text)
        logger.info(f"[Extract] JSON parse succeeded after cleaning")
    except json.JSONDecodeError:
        # 方案2: 提取多个JSON对象
        logger.warning(f"[Extract] Trying multi-object parse")
        objects = extract_json_objects(response_text)

        if objects:
            # 合并所有destinations
            result = {"destinations": []}
            for obj in objects:
                if isinstance(obj, dict) and "destinations" in obj:
                    result["destinations"].extend(obj["destinations"])
            logger.info(f"[Extract] Multi-object parse succeeded: {len(result['destinations'])} destinations")
        else:
            logger.warning(f"[Extract] JSON parse failed for chunk {i}, skipping")
            logger.debug(f"[Extract] Raw response: {response_text[:500]}")
            continue
```
