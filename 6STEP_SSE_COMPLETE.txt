### 选项2：前后端SSE交互标准

**SSE事件格式（匹配前端WorkflowProgress组件）：**

#### 1. start 事件
```json
event: start
data: {
  "message": "开始处理文档: xxx.pdf",
  "filename": "xxx.pdf"
}
```

#### 2. progress 事件（每个步骤开始时）
```json
event: progress
data: {
  "step": "minio",
  "step_name": "MinIO存储",
  "status": "processing",
  "message": "正在保存文件到MinIO存储...",
  "progress": 15
}
```

#### 3. step_complete 事件（每个步骤完成时）
```json
event: step_complete
data: {
  "step": "minio",
  "step_name": "MinIO存储",
  "status": "completed",
  "message": "文件保存成功",
  "progress": 15
}
```

#### 4. complete 事件（全部完成）
```json
event: complete
data: {
  "message": "处理完成: 提取了 8 个景点",
  "destinations_count": 8,
  "progress": 100
}
```

#### 5. error 事件（出错时）
```json
event: error
data: {
  "message": "处理失败: 错误信息",
  "error": "详细错误"
}
```

### 前端监听代码

```javascript
// WorkflowProgress.vue
const connectSSE = () => {
  const url = `/api/documents/progress/${props.taskId}`
  eventSource = new EventSource(url)

  // 监听 start 事件
  eventSource.addEventListener('start', (event) => {
    const data = JSON.parse(event.data)
    statusText.value = data.message
    addLog(data.message)
  })

  // 监听 progress 事件（步骤开始）
  eventSource.addEventListener('progress', (event) => {
    const data = JSON.parse(event.data)
    progress.value = data.progress
    updateStepStatus(data.step, 'processing')
    statusText.value = data.message
  })

  // 监听 step_complete 事件（步骤完成）
  eventSource.addEventListener('step_complete', (event) => {
    const data = JSON.parse(event.data)
    updateStepStatus(data.step, 'completed')
  })

  // 监听 complete 事件（全部完成）
  eventSource.addEventListener('complete', (event) => {
    const data = JSON.parse(event.data)
    completed.value = true
    progress.value = 100
    statusText.value = '处理完成！'
    emit('complete')
    eventSource.close()
  })

  // 监听 error 事件
  eventSource.addEventListener('error', (event) => {
    const data = JSON.parse(event.data)
    errorMessage.value = data.message
    eventSource.close()
  })
}
```
