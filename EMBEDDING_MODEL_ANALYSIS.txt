### 方案1: 创建Embedding服务

**创建 `app/core/embedding_service.py`：**
```python
from sentence_transformers import SentenceTransformer
from typing import List
from loguru import logger
from config.settings import settings

class EmbeddingService:
    """本地Embedding模型服务"""

    def __init__(self):
        self.model_path = settings.EMBEDDING_MODEL
        self.model = None

    def load_model(self):
        """加载本地模型"""
        if self.model is None:
            logger.info(f"[Embedding] Loading model: {self.model_path}")
            self.model = SentenceTransformer(self.model_path)
            logger.success(f"[Embedding] Model loaded successfully")

    def encode(self, texts: List[str]) -> List[List[float]]:
        """生成embedding向量"""
        if self.model is None:
            self.load_model()

        embeddings = self.model.encode(texts, convert_to_numpy=True)
        return embeddings.tolist()

# 单例
_embedding_service = None

def get_embedding_service():
    global _embedding_service
    if _embedding_service is None:
        _embedding_service = EmbeddingService()
        _embedding_service.load_model()
    return _embedding_service
```

### 方案2: 在向量化节点中使用Embedding服务

**修改 `node_vectorization.py` 的 `vectorize_traditional()` 函数：**
```python
async def vectorize_traditional(state: DocumentProcessingState):
    logger.info("[Vector] Traditional RAG vectorization with Milvus")

    try:
        destinations = state.get('destinations', [])

        if not destinations:
            logger.warning("[Vector] No destinations to vectorize")
            return state

        # 1. 加载embedding服务
        from app.core.embedding_service import get_embedding_service
        embedding_service = get_embedding_service()

        # 2. 准备文本
        texts = []
        for dest in destinations:
            text = f"{dest.get('name', '')} {dest.get('description', '')} {dest.get('city', '')} {dest.get('province', '')}"
            texts.append(text)

        # 3. 生成embedding向量
        logger.info(f"[Vector] Generating embeddings for {len(texts)} texts")
        embeddings = embedding_service.encode(texts)

        # 4. 插入Milvus
        from app.core.milvus_client import get_milvus_client
        milvus = get_milvus_client()

        data = []
        for i, dest in enumerate(destinations):
            data.append({
                "destination_id": dest.get('graph_id', f"dest_{i}"),
                "embedding": embeddings[i],
                "text": texts[i]
            })

        milvus.insert(data)

        state['vector_success'] = True
        logger.success(f"[Vector] Milvus vectorization: {len(destinations)} destinations added")

    except Exception as e:
        logger.error(f"[Vector] Vectorization failed: {e}")
        state['vector_success'] = False

    return state
```
